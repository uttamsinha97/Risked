#include <iostream>
using namespace std;
long int countSquares(int n)
{
    return (n * (n + 1) / 2) * (2*n + 1) / 3;
}
 
int main()
{
    int t,i,n;
  scanf("%d",&t);
  for(i=1;i<=t;i++)
  {
        scanf("%d",&n);
        cout<< countSquares(n)<<endl;
  }
  return 0;
}
